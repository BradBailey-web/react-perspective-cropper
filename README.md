```bash

```

or

```bash
yarn add react-doc-scan
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-doc-scan'
import 'react-doc-scan/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
